515  
516          # __doc__ (as of 2008-08-02) for pygame.display.set_gamma_ramp:
517  
518          # change the hardware gamma ramps with a custom lookup
519          # pygame.display.set_gamma_ramp(red, green, blue): return bool
520          # set_gamma_ramp(red, green, blue): return bool
521          #
522          # Set the red, green, and blue gamma ramps with an explicit lookup
523          # table. Each argument should be sequence of 256 integers. The
524          # integers should range between 0 and 0xffff. Not all systems and
525          # hardware support gamma ramps, if the function succeeds it will
526          # return True.
527          #
528          pygame.display.set_mode((5, 5))
529          r = list(range(256))
530          g = [number + 256 for number in r]
531          b = [number + 256 for number in g]
532          isSupported = pygame.display.set_gamma_ramp(r, g, b)
533          if (isSupported):
534              self.assertTrue(pygame.display.set_gamma_ramp(r, g, b))
535          else:
536              self.assertFalse(pygame.display.set_gamma_ramp(r, g, b))

604          """Test for toggle fullscreen"""
605  
606          #try to toggle fullscreen with no active display
607          #this should result in an error
608          pygame.display.quit()
609          with self.assertRaises(pygame.error):
610              pygame.display.toggle_fullscreen()
611  
612          pygame.display.init()
613          width_height = (640,480)
614          test_surf = pygame.display.set_mode(width_height)
615  
616          #try to toggle fullscreen
617          try:
618              pygame.display.toggle_fullscreen()
619  
620          except pygame.error:
621              self.fail()
622  
623          else:
624              #if toggle success, the width/height should be a
625              #value found in list_modes
626              if pygame.display.toggle_fullscreen() == 1:
627                  boolean = (test_surf.get_width(), test_surf.get_height()) \
628                  in pygame.display.list_modes(depth=0, flags=pygame.FULLSCREEN, display=0)
629  
630                  self.assertEqual(boolean, True)
631  
632              #if not original width/height should be preserved
633              else:
634                  self.assertEqual((test_surf.get_width(), test_surf.get_height()), \
635                  width_height)
