28       # Represent escaped bytes and strings in a portable way.
29       #
30       # as_bytes: Allow a Python 3.x string to represent a bytes object.
31       #   e.g.: as_bytes("a\x01\b") == b"a\x01b" # Python 3.x
32       #         as_bytes("a\x01\b") == "a\x01b"  # Python 2.x
33       # as_unicode: Allow a Python "r" string to represent a unicode string.
34       #   e.g.: as_unicode(r"Bo\u00F6tes") == u"Bo\u00F6tes" # Python 2.x
35       #         as_unicode(r"Bo\u00F6tes") == "Bo\u00F6tes"  # Python 3.x
36       def as_bytes(string):
37           """ '<binary literal>' => b'<binary literal>' """
38           return string.encode('latin-1', 'strict')
39   
40       def as_unicode(rstring):
41           """ r'<Unicode literal>' => '<Unicode literal>' """
42           return rstring.encode('ascii', 'strict').decode('unicode_escape',
43                                                           'strict')

96   def filesystem_encode(u):
97       fsencoding = sys.getfilesystemencoding()
98       if fsencoding.lower() in ['ascii', 'ansi_x3.4-1968'] and sys.platform.startswith('linux'):
99           # Don't believe Linux systems claiming ASCII-only filesystems. In
100          # practice, arbitrary bytes are allowed, and most things expect UTF-8.
101          fsencoding = 'utf-8'
102      return u.encode(fsencoding, filesystem_errors)
