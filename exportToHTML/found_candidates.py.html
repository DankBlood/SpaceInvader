found_candidates.py 
1    """Utilities to lazily create and visit candidates found. 
2     
3    Creating and visiting a candidate is a *very* costly operation. It involves 
4    fetching, extracting, potentially building modules from source, and verifying 
5    distribution metadata. It is therefore crucial for performance to keep 
6    everything here lazy all the way down, so we only touch candidates that we 
7    absolutely need, and not "download the world" when we only need one version of 
8    something. 
9    """
10   
11   import functools
12   
13   from pip._vendor.six.moves import collections_abc  # type: ignore
14   
15   from pip._internal.utils.typing import MYPY_CHECK_RUNNING
16   
17   if MYPY_CHECK_RUNNING:
18       from typing import Callable, Iterator, Optional, Set, Tuple
19   
20       from pip._vendor.packaging.version import _BaseVersion
21   
22       from .base import Candidate
23   
24       IndexCandidateInfo = Tuple[_BaseVersion, Callable[[], Optional[Candidate]]]
25

27   def _iter_built(infos):
28       # type: (Iterator[IndexCandidateInfo]) -> Iterator[Candidate]
29       """Iterator for ``FoundCandidates``. 
30    
31       This iterator is used the package is not already installed. Candidates 
32       from index come later in their normal ordering. 
33       """
34       versions_found = set()  # type: Set[_BaseVersion]
35       for version, func in infos:
36           if version in versions_found:
37               continue
38           candidate = func()
39           if candidate is None:
40               continue
41           yield candidate
42           versions_found.add(version)
43   
44   
45   def _iter_built_with_prepended(installed, infos):
46       # type: (Candidate, Iterator[IndexCandidateInfo]) -> Iterator[Candidate]
47       """Iterator for ``FoundCandidates``. 
48    
49       This iterator is used when the resolver prefers the already-installed 
50       candidate and NOT to upgrade. The installed candidate is therefore 
51       always yielded first, and candidates from index come later in their 
52       normal ordering, except skipped when the version is already installed. 
53       """
54       yield installed
55       versions_found = {installed.version}  # type: Set[_BaseVersion]
56       for version, func in infos:
57           if version in versions_found:
58               continue
59           candidate = func()
60           if candidate is None:
61               continue
62           yield candidate
63           versions_found.add(version)
64   
65   
66   def _iter_built_with_inserted(installed, infos):
67       # type: (Candidate, Iterator[IndexCandidateInfo]) -> Iterator[Candidate]
68       """Iterator for ``FoundCandidates``. 
69    
70       This iterator is used when the resolver prefers to upgrade an 
71       already-installed package. Candidates from index are returned in their 
72       normal ordering, except replaced when the version is already installed. 
73    
74       The implementation iterates through and yields other candidates, inserting 
75       the installed candidate exactly once before we start yielding older or 
76       equivalent candidates, or after all other candidates if they are all newer. 
77       """
78       versions_found = set()  # type: Set[_BaseVersion]
79       for version, func in infos:
80           if version in versions_found:
81               continue
82           # If the installed candidate is better, yield it first.
83           if installed.version >= version:
84               yield installed
85               versions_found.add(installed.version)
86           candidate = func()
87           if candidate is None:
88               continue
89           yield candidate
90           versions_found.add(version)
91   
92       # If the installed candidate is older than all other candidates.
93       if installed.version not in versions_found:
94           yield installed
95

97   class FoundCandidates(collections_abc.Sequence):
98       """A lazy sequence to provide candidates to the resolver. 
99    
100      The intended usage is to return this from `find_matches()` so the resolver 
101      can iterate through the sequence multiple times, but only access the index 
102      page when remote packages are actually needed. This improve performances 
103      when suitable candidates are already installed on disk. 
104      """
105      def __init__(
106          self,
107          get_infos,  # type: Callable[[], Iterator[IndexCandidateInfo]]
108          installed,  # type: Optional[Candidate]
109          prefers_installed,  # type: bool
110      ):
111          self._get_infos = get_infos
112          self._installed = installed
113          self._prefers_installed = prefers_installed
114  
115      def __getitem__(self, index):
116          # type: (int) -> Candidate
117          # Implemented to satisfy the ABC check. This is not needed by the
118          # resolver, and should not be used by the provider either (for
119          # performance reasons).
120          raise NotImplementedError("don't do this")
121  
122      def __iter__(self):
123          # type: () -> Iterator[Candidate]
124          infos = self._get_infos()
125          if not self._installed:
126              return _iter_built(infos)
127          if self._prefers_installed:
128              return _iter_built_with_prepended(self._installed, infos)
129          return _iter_built_with_inserted(self._installed, infos)
130  
131      def __len__(self):
132          # type: () -> int
133          # Implemented to satisfy the ABC check. This is not needed by the
134          # resolver, and should not be used by the provider either (for
135          # performance reasons).
136          raise NotImplementedError("don't do this")
137  
138      @functools.lru_cache(maxsize=1)
139      def __bool__(self):
140          # type: () -> bool
141          if self._prefers_installed and self._installed:
142              return True
143          return any(self)
144  
145      __nonzero__ = __bool__  # XXX: Python 2.
