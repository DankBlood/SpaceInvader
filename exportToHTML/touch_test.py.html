40   class TouchInteractiveTest(unittest.TestCase):
41   
42       __tags__ = ["interactive"]
43   
44       @unittest.skipIf(not has_touchdevice, "no touch devices found")
45       def test_get_finger(self):
46           """ask for touch input and check the dict"""
47   
48           pygame.display.init()
49           pygame.font.init()
50   
51           os.environ['SDL_VIDEO_WINDOW_POS'] = '50,50'
52           screen = pygame.display.set_mode((800, 600))
53           screen.fill((255, 255, 255))
54   
55           font = pygame.font.Font(None, 32)
56           instructions_str_1 = 'Please place some fingers on your touch device'
57           instructions_str_2 = ('Close the window when finished, '
58                                 'and answer the question')
59           inst_1_render = font.render(instructions_str_1,
60                                       True, pygame.Color('#000000'))
61           inst_2_render = font.render(instructions_str_2,
62                                       True, pygame.Color('#000000'))
63   
64           running = True
65           while running:
66               for event in pygame.event.get():
67                   if event.type == pygame.QUIT:
68                       running = False
69   
70               finger_data_renders = []
71               num_devices = pygame._sdl2.touch.get_num_devices()
72               if num_devices > 0:
73                   first_device = pygame._sdl2.touch.get_device(0)
74                   num_fingers = pygame._sdl2.touch.get_num_fingers(first_device)
75                   if num_fingers > 0:
76                       for finger_index in range(0, num_fingers):
77                           data = pygame._sdl2.touch.get_finger(first_device,
78                                                                finger_index)
79                           render = font.render('finger - ' + str(data),
80                                                True, pygame.Color('#000000'))
81   
82                           finger_data_renders.append(render)
83   
84               screen.fill((255, 255, 255))
85               screen.blit(inst_1_render, (5, 5))
86               screen.blit(inst_2_render, (5, 40))
87               for index, finger in enumerate(finger_data_renders):
88                   screen.blit(finger, (5, 80 + (index * 40)))
89   
90               pygame.display.update()
91   
92           response = question("Does the finger data seem correct?")
93           self.assertTrue(response)
94   
95           pygame.display.quit()
96
