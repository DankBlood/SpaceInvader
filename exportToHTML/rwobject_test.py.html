113      def test_pathlib_obj(self):
114          """Test loading string representation of pathlib object"""
115          """ 
116          We do this because pygame functions internally use pg_EncodeString 
117          to decode the filenames passed to them. So if we test that here, we 
118          can safely assume that all those functions do not have any issues 
119          with pathlib objects 
120          """
121          encoded = encode_string(pathlib.PurePath("foo"), "utf-8")
122          self.assertEqual(encoded, b"foo")

124          encoded = encode_string(pathlib.Path("baz"))
